\section*{Standard C}

\subsection*{Evoluzione Standard}

\begin{description}
    \item[C89/C90 (ANSI C)] Prima standardizzazione (1989/1990)
    \begin{itemize}
        \item Standard library (stdio.h, stdlib.h, string.h)
        \item Function prototypes
        \item \texttt{void} pointer
    \end{itemize}

    \item[C99 (ISO/IEC 9899:1999)] Aggiornamento maggiore
    \begin{itemize}
        \item Inline functions
        \item Variable-length arrays (VLA)
        \item \texttt{//} comments (stile C++)
        \item \texttt{long long int}, \texttt{\_Bool}, \texttt{\_Complex}
        \item \texttt{<stdint.h>}, \texttt{<stdbool.h>}
        \item Designated initializers: \texttt{struct point p = \{.x=1, .y=2\};}
    \end{itemize}

    \item[C11 (ISO/IEC 9899:2011)] Focus su concorrenza e sicurezza
    \begin{itemize}
        \item Multi-threading: \texttt{<threads.h>}, \texttt{\_Thread\_local}
        \item Atomic operations: \texttt{<stdatomic.h>}
        \item Static assertions: \texttt{\_Static\_assert}
        \item Anonymous structures/unions
        \item \texttt{\_Generic} (type-generic macros)
        \item Bounds-checking functions: \texttt{*\_s} variants (optional)
    \end{itemize}

    \item[C17/C18 (ISO/IEC 9899:2018)] Bug fixes e clarifications
    \begin{itemize}
        \item Nessuna feature nuova, solo correzioni C11
    \end{itemize}

    \item[C23 (ISO/IEC 9899:2023)] Standard più recente (2023)
    \begin{itemize}
        \item \texttt{\_BitInt(N)} arbitrary-width integers
        \item \texttt{typeof}, \texttt{\_BitInt}
        \item \texttt{constexpr}
        \item Digit separators: \texttt{1'000'000}
        \item Binary literals: \texttt{0b10101010}
        \item \texttt{nullptr}
    \end{itemize}
\end{description}

    \item \textbf{Scrivi codice quotidianamente}: Almeno 30 minuti al giorno
    \item \textbf{Risolvi problemi}: HackerRank, LeetCode (20+ problemi)
    \item \textbf{Usa Valgrind}: Per ogni programma, controlla memory leak
    \item \textbf{Leggi codice}: Studia progetti open-source (Git, Redis)
    \item \textbf{Compila con warning}: \texttt{gcc -Wall -Wextra -Werror}
\end{enumerate}

\subsection*{Progetti Suggeriti}

\begin{enumerate}
    \item \textbf{Calcolatrice CLI}: Input, parsing, eval espressioni
    \item \textbf{Todo List}: File I/O, struct, dynamic memory
    \item \textbf{Text Editor}: Buffer manipulation, file handling
    \item \textbf{Shell Minimale}: fork, exec, pipe, redirection
    \item \textbf{Web Server HTTP}: Socket programming, HTTP parsing
    \item \textbf{Database Key-Value}: Hash table, file persistence
    \item \textbf{Compression Tool}: Algoritmo Huffman coding
    \item \textbf{Malloc Implementation}: Custom memory allocator
\end{enumerate}

\subsection*{Errori Comuni da Evitare}

\begin{itemize}
    \item \textbf{Buffer overflow}: Usa \texttt{strncpy} invece di \texttt{strcpy}
    \item \textbf{Memory leak}: Ogni \texttt{malloc} deve avere un \texttt{free}
    \item \textbf{Dangling pointer}: Non usare puntatori dopo \texttt{free}
    \item \textbf{Undefined behavior}: Evita array[-1], x/0, uninitialized vars
    \item \textbf{Portabilità}: Usa \texttt{<stdint.h>} (int32\_t invece di int)
    \item \textbf{Warning ignorati}: Risolvi TUTTI i warning del compilatore
\end{itemize}

\section*{Roadmap Apprendimento C}

\subsection*{Livello Principiante (1-3 mesi)}

\begin{itemize}
    \item Sintassi base: variabili, operatori, control flow
    \item Funzioni e scope
    \item Array e stringhe
    \item Struct
    \item File I/O (fopen, fread, fwrite)
    \item \textbf{Progetto}: Calcolatrice, gestionale studenti
\end{itemize}

\subsection*{Livello Intermedio (3-6 mesi)}

\begin{itemize}
    \item Puntatori avanzati (double pointer, function pointer)
    \item Dynamic memory (malloc, calloc, realloc, free)
    \item Preprocessing (\#define, \#include, conditional compilation)
    \item Makefile
    \item Debugging (GDB, Valgrind)
    \item \textbf{Progetto}: Editor di testo, shell minimale
\end{itemize}

\subsection*{Livello Avanzato (6-12 mesi)}

\begin{itemize}
    \item System programming (fork, exec, signals, IPC)
    \item Networking (socket, TCP/UDP)
    \item Multithreading (pthread, mutex, semaphore)
    \item Librerie dinamiche (.so, .dll)
    \item Profiling e ottimizzazione
    \item \textbf{Progetto}: Web server, chat multithread, database embedded
\end{itemize}
